180
181 for(int stack = 0; stack <= stacks; ++stack){
182 float v = (float)stack / (float)stacks;
183 float phi = v * (float)Math3D::PI;
184 float y = std::cos(phi);
185 float ringRadius = std::sin(phi);
186
187 for(int slice = 0; slice <= slices; ++slice){
188 float u = (float)slice / (float)slices;
189 float theta = u * (float)Math3D::PI * 2.0f;
190 float x = ringRadius * std::cos(theta);
191 float z = ringRadius * std::sin(theta);
192
193 Math3D::Vec3 normal(x, y, z);
