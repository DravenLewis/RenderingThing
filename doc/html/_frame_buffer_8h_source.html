49 // 1. Create the Depth Texture
50 // We do this manually here (or you could add a helper in Texture.h)
51 depthTexture = std::make_shared<Texture>();
52 // Note: We don't set width/height on the Texture object here manually
53 // unless your Texture class requires it.
54
55 GLuint tid;
56 glGenTextures(1, &tid);
57 depthTexture->getID() = tid;
58
59 // DEBUG CHECK
60 if(depthTexture->getID() == 0) {
61 LogBot.Log(LOG_FATL, "[FrameBuffer] CRITICAL: glGenTextures failed for Depth Texture!");
62 }
63
64 glBindTexture(GL_TEXTURE_2D, depthTexture->getID());
65
66 // GL_DEPTH_COMPONENT24 is standard for precision
67 glTexImage2D(GL_TEXTURE_2D, 0, GL_DEPTH_COMPONENT24, width, height, 0, GL_DEPTH_COMPONENT, GL_FLOAT, NULL);
68
69 // Essential Texture Parameters for Depth
70 // Linear filtering keeps post-process depth lookups (SSAO/DOF) stable across pixel edges.
71 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
72 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
73 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_BORDER);
74 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_BORDER);
75
76 // Border color for depth (white = far away) prevents artifacts at screen edges
77 float borderColor[] = { 1.0f, 1.0f, 1.0f, 1.0f };
78 glTexParameterfv(GL_TEXTURE_2D, GL_TEXTURE_BORDER_COLOR, borderColor);
79
80 // 2. Attach Depth Texture to FBO
81 bind();
82 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_DEPTH_ATTACHMENT, GL_TEXTURE_2D, depthTexture->getID(), 0);
83
84 // Note: We are currently NOT complete because we have no Color buffer yet.
85 // Screen class attaches color later via attachTexture().
86 // If you want to use this for Shadow Mapping (Depth Only), you need to tell GL to draw no color:
87 // glDrawBuffer(GL_NONE); glReadBuffer(GL_NONE);
88
89 // Crucial: Tell OpenGL we aren't drawing to color YET
90 glDrawBuffer(GL_NONE);
91 glReadBuffer(GL_NONE);
92
93 unbind();
94 }

158 void attachTexture(PTexture tex){
159
160 this->texturePtr = tex;
161
162 bind();
163 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, tex->getID(), 0);
164
165 GLenum drawBuffers[1] = {GL_COLOR_ATTACHMENT0};
166 glDrawBuffers(1, drawBuffers);
167 glReadBuffer(GL_COLOR_ATTACHMENT0);
168
169 GLenum status = glCheckFramebufferStatus(GL_FRAMEBUFFER);
170 if (status != GL_FRAMEBUFFER_COMPLETE) {
171 std::string error;
172 switch (status) {
173 case GL_FRAMEBUFFER_INCOMPLETE_ATTACHMENT: error = "Incomplete Attachment"; break;
174 case GL_FRAMEBUFFER_INCOMPLETE_MISSING_ATTACHMENT: error = "Missing Attachment"; break;
175 case GL_FRAMEBUFFER_UNSUPPORTED: error = "Unsupported Format"; break;
176 default: error = "Unknown Error"; break;
177 }
178 LogBot.Log(LOG_ERRO, "[FrameBuffer] FBO Error: %s (Code: %u)", error.c_str(), status);
179 }
180
181 unbind();
182 }

227 static std::shared_ptr<FrameBuffer> CreateGBuffer(int width, int height){
228 auto fbuffer = std::make_shared<FrameBuffer>(width, height);
229
230 fbuffer->gbufferAttachments.clear();
231 fbuffer->gbufferAttachments.reserve(3);
232
233 auto createAttachment = [&](GLenum internalFormat, GLenum format, GLenum type) -> PTexture {
234 GLuint texId = 0;
235 glGenTextures(1, &texId);
236 glBindTexture(GL_TEXTURE_2D, texId);
237 glTexImage2D(GL_TEXTURE_2D, 0, internalFormat, width, height, 0, format, type, NULL);
238 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
239 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
240 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_S, GL_CLAMP_TO_EDGE);
241 glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_WRAP_T, GL_CLAMP_TO_EDGE);
242 glBindTexture(GL_TEXTURE_2D, 0);
243 return Texture::CreateFromExisting(texId, width, height, true);
244 };
245
246 fbuffer->bind();
247
248 {
249 GBufferAttachment albedo;
250 albedo.internalFormat = GL_RGBA8;
251 albedo.format = GL_RGBA;
252 albedo.type = GL_UNSIGNED_BYTE;
253 albedo.texture = createAttachment(albedo.internalFormat, albedo.format, albedo.type);
254 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT0, GL_TEXTURE_2D, albedo.texture->getID(), 0);
255 fbuffer->gbufferAttachments.push_back(albedo);
256 }
257
258 {
259 GBufferAttachment normal;
260 normal.internalFormat = GL_RGBA16F;
261 normal.format = GL_RGBA;
262 normal.type = GL_FLOAT;
263 normal.texture = createAttachment(normal.internalFormat, normal.format, normal.type);
264 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT1, GL_TEXTURE_2D, normal.texture->getID(), 0);
265 fbuffer->gbufferAttachments.push_back(normal);
266 }

269 GBufferAttachment position;
270 // Large receivers (e.g. 200x200 ground planes) need full precision to avoid deferred-light banding.
271 position.internalFormat = GL_RGBA32F;
272 position.format = GL_RGBA;
273 position.type = GL_FLOAT;
274 position.texture = createAttachment(position.internalFormat, position.format, position.type);
275 glFramebufferTexture2D(GL_FRAMEBUFFER, GL_COLOR_ATTACHMENT2, GL_TEXTURE_2D, position.texture->getID(), 0);
276 fbuffer->gbufferAttachments.push_back(position);
277 }

280 glDrawBuffers(3, drawBuffers);

291 }
