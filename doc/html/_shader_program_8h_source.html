302 Shader _createShader(std::string glsl, ShaderType type){
303 Shader value = 0;
304
305 const char * str = glsl.c_str();
306
307 switch (type){
308 case ShaderType::VERTEX:
309 value = glCreateShader(GL_VERTEX_SHADER);
310 break;
311 case ShaderType::FRAGMENT:
312 value = glCreateShader(GL_FRAGMENT_SHADER);
313 break;
314 case ShaderType::GEOMETRY:
315 value = glCreateShader(GL_GEOMETRY_SHADER);
316 break;
317 case ShaderType::TESSELATION:
318 value = glCreateShader(GL_TESS_CONTROL_SHADER);
319 break;
320 case ShaderType::COMPUTE:
321 value = glCreateShader(GL_COMPUTE_SHADER);
322 break;
323 case ShaderType::TASK:
324 value = glCreateShader(GL_TESS_EVALUATION_SHADER);
325 break;
326 case ShaderType::RAYTRACE:
327 default:
328 value = 0;
329 break;
330 }
331
332 if(value != 0){
333 glShaderSource(value,1,&str, NULL);
334 glCompileShader(value);
335 }
336
337 return value;
338 }

514 std::shared_ptr<ShaderProgram> getOrCompile(
515 std::string name,
516 std::string vtx = "",
517 std::string frag = "",
518 std::string geom = "",
519 std::string tess = "",
520 std::string compute = "",
521 std::string task = "",
522 std::string srtx = ""
523 ){
524 // Check if the file is in the cache;
525 auto it = programCache.find(name);
526 if(it != programCache.end()){
527 if(it->second && it->second->getID() != 0){
528 return it->second;
529 }
530 // Recover from failed cached programs (ID==0) by recompiling.
531 programCache.erase(it);
532 }
533
534 auto program = std::make_shared<ShaderProgram>();
535 if(vtx.size() != 0) program->setVertexShader(vtx);
536 if(frag.size() != 0) program->setFragmentShader(frag);
537 if(geom.size() != 0) program->setGeometryShader(geom);
538 if(tess.size() != 0) program->setTesselationShader(tess);
539 if(compute.size() != 0) program->setComputeShader(compute);
540 if(task.size() != 0) program->setTaskShader(task);
541 if(srtx.size() != 0) program->setRTXShader(srtx);
542
543
544 program->compile();
545
546 programCache[name] = program;
547
548 return program;
549 }
550
554 void clear(){
555 programCache.clear();
556 }
557};
